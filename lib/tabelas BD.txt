CREATE TABLE jaulas
	(idJaula serial,
	 modelo int,
	 descricao varchar(500),
	 animalid int,
	 primary key(idJaula));
	 
CREATE TABLE animais
	(idAnimal serial,
	peso float,
	membros int,
	idade int,
	nome varchar(30),
	jaulaId int,
	primary key(idAnimal));
	
ALTER TABLE jaulas
ADD FOREIGN KEY (animalid) REFERENCES animais(idAnimal) ON DELETE CASCADE;

ALTER TABLE animais
ADD FOREIGN KEY (jaulaId) REFERENCES jaulas(idJaula) ON DELETE CASCADE;
